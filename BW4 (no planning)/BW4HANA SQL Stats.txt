select 'RSPMPROCESS1', SUBSTR_BEFORE(SRC,'  '), max_time, SUBSTR_AFTER(SRC,'  '), '' from "RSBKDTP"
left outer join
( select PROCESS_ID, max(PROCESS_TSN) as max_time from "RSPMPROCESS" group by PROCESS_ID, PROCESS_TYPE HAVING PROCESS_TYPE = 'DTP_LOAD')
as A on trim(DTP) = trim(PROCESS_ID)
where OBJVERS = 'A' and SRCTP = 'DTASRC'
union
select 'RSPMPROCESS2', target, max(max_time), '', ''
from
(
    select trim(SUBSTR_AFTER(SRC,'  ')) as target, max_time,  '' from "RSBKDTP"
    left outer join
    ( select PROCESS_ID, max(PROCESS_TSN) as max_time from "RSPMPROCESS" group by PROCESS_ID, PROCESS_TYPE HAVING PROCESS_TYPE = 'DTP_LOAD')
    as A on DTP = PROCESS_ID
    where OBJVERS = 'A' and SRCTP = 'DTASRC'
) group by target
